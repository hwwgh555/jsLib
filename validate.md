## 校验规则
/**
* 通用正则
*/
```
const RegExp = {
    regCnEnNum: /^[\u4e00-\u9fa5_a-zA-Z0-9]+$/, // 汉字+字母+数字
    regEnNum: /^[0-9a-zA-Z]+$/, // 字母+数字
    exceptNum: /[^0-9]/ig, // 除了数字其他
    regNum: /^\d+(\.\d+)?$/, // 整数+小数
    enNumPeriod: /^[0-9a-zA-Z\.]+$/, // 字母+数字＋.如版本号
    num: /^(?!(0[0-9]{0,}$))[0-9]{1,}[.]{0,}[0-9]{0,}$/, // 大于０的数字
    limitNum: /^(([0-9]\d{0,11}(?:|\.\d{0,}))|(0\.\d{0,}))$/, // 大于0.0000001 小于100000000 以０.开头小数点后７位
}
export default RegExp
```
